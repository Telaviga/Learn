gcc (имя файла компиляции) (имя выходного файла компияции с расширением)
./ - запуск скомпилированног файла
#include - включение в программу стандартного или определяемого пользователем файла
<stdio.h> - стандартный заголовочный файл ввода и вывода
main - это точка входа приложения С
char - один байт в котором сохраняется один символ (пример) 'A'
int - целое десятичное число (пример) 10
float - число с плавающей точкой (точность до 6 знаков) (пример) 0.123456
double - число с плавающей точкой (точность до 10 знаков) (пример) 0.123456789
short - короткое целое(2 байта) (пример) 10 100
long - большое целое(4 байта)
long long - огромное целое (8 байт)
'' - буквенный символ в ковычках 
"" - строка в ковычках
; - конец любой строки
// - комментарий в строку
/* */ - многострочный комментарий
